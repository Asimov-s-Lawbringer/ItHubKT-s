2)Создание Процедуры

DELIMITER //

CREATE PROCEDURE Transaction (
IN sender_acc VARCHAR(20), -- сюда номер аккаунта отправиетеля
IN reciever_acc VARCHAR(20),-- сюда получателя (аккаунты связаны внешним ключом)
IN amount DECIMAL(5,2) -- сюда сколько средствр передавать
)

BEGIN
	DECLARE sender_balance DECIMAL(5, 2); -- переменная для проверок

	START TRANSACTION;-- все что после можно отменить при ошибке

	SELECT balance INTO sender_balance FROM Accounts -- берем количество денег и записываем в баланс отправителя
	WHERE account_number = sender_acc FOR UPDATE; -- где номер счета == указанному номеру счета

	IF sender_balance >= amount THEN -- если денег хватает то начинаем перевод
		UPDATE Accounts SET balance = balance - amount -- списываем
		WHERE account_number = sender_acc -- у отправителя

		UPDATE Accounts SET balance = balance + amount -- прибавляем
		WHERE account_number = reciever_acc -- получателю

		COMMIT; -- подтверждаем операцию
	ELSE -- или если денег недостаточно
		ROLLBACK; -- откатываем все как было до STAR T TRANSACtion
		SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT= 'Сумма отправки больше чем имеющиеся средства'
	ENDIF;
END//

def make_transaction(request):
    if request.method == "POST":
        connection = data_base_connector()
        sender_acc = request.POST.get("sender_acc")
        receiver_acc = request.POST.get("receiver_acc")
        amount = request.POST.get("amount")
        try:
            with connection.cursor() as cursor:
                cursor.callproc('MakeTransaction', [sender_acc,receiver_acc,amount])
        except:
            print("Ошибка в передаче")
        finally:
            connection.commit()
            connection.close()
    return render(request,'make_transaction.html')


def user_login(request):
    if request.method == "POST":
        connection = data_base_connector()
        login_name = request.POST.get("user_name")
        phone = request.POST.get("user_phone")

        try:
            with connection.cursor() as cursor:
                # ищем юзера
                sql = "SELECT name FROM Users WHERE name = %s AND phone = %s;"
                cursor.execute(sql, (login_name, phone))

                # читаем
                result = cursor.fetchone()

            # смотрим на строку
            if result:
                name_from_db = result['name'] if isinstance(result, dict) else result[0]
                if name_from_db == login_name:
                    print("Вы вошли")
                    #return redirect(' ')  перенаправляем (пока не готово)

            # Если юзер не найден
            print("Неверное имя или телефон")
            return render(request, 'login.html', {'error': 'Неверные данные'})

        except Exception as e:
            print(f"Ошибка базы данных: {e}")
            return render(request, 'login.html', {'error': 'Ошибка сервера'})

        finally:
            connection.close()
    return render(request, 'login.html')
